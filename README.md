# minimal nvim config

keybinds

- space is leader key
- <c-h/j/k/l> - window nav
- ctrl-a - select all
- esc - clear search
- space+sf - find files
- space+sg - grep files
- space+/ - find in buffer
- [c]/]c - prev/next git change
- space+hr/hp/hb - reset/preview/blame git hunk
- gd - goto definition
- gr - references
- K - docs
- space+rn - rename
- space+ca - code action
